** Dropbox Path Configuration

Dropbox 경로를 한 곳에서 관리한다. 플랫폼에 따라 기본 경로가 다를 수 있으므로 필요 시 =my/dropbox-path= 변수를 변경하면 된다.

#+begin_src emacs-lisp
  (defvar my/dropbox-path
    (cond
     ((eq system-type 'windows-nt)
      (or (getenv "DROPBOX_PATH")
          (expand-file-name "~/Dropbox")))
     (t (expand-file-name "~/Dropbox")))
    "Base path to Dropbox directory.
On Windows, you can set DROPBOX_PATH environment variable to override.")

  (defun my/dropbox-expand (path)
    "Expand PATH relative to `my/dropbox-path'."
    (expand-file-name path my/dropbox-path))

  (setq org-directory (expand-file-name "~/org"))

  (defun my/org-expand (path)
    "Expand PATH relative to `org-directory'."
    (expand-file-name path org-directory))
#+end_src

#+begin_src emacs-lisp
  (defun org-babel-execute:json (body params)
    (let ((jq (cdr (assoc :jq params)))
          (node (cdr (assoc :node params))))
      (cond
       (jq
        (unless (executable-find "jq")
          (user-error "jq command not found. Please install jq"))
        (with-temp-buffer
          ;; Insert the JSON into the temp buffer
          (insert body)
          ;; Run jq command on the whole buffer using call-process-region
          ;; This avoids shell injection by passing jq filter as a direct argument
          (let ((exit-code (call-process-region
                           (point-min) (point-max)
                           "jq" t t nil "-r" jq)))
            (unless (zerop exit-code)
              (error "jq command failed with exit code %d" exit-code)))
          ;; Return the contents of the temp buffer as the result
          (buffer-string)))
       (node
        (unless (executable-find "node")
          (user-error "node command not found. Please install Node.js"))
        (with-temp-buffer
          (insert (format "const it = %s;" body))
          (insert node)
          (let ((exit-code (call-process-region
                           (point-min) (point-max)
                           "node" t t nil "-p")))
            (unless (zerop exit-code)
              (error "node command failed with exit code %d" exit-code)))
          (buffer-string))))))
#+end_src

** Tree-Sitter

현재 emacs 최신 버전(emacs-29) 공식 릴리즈에서는 tree-siter를 기본적으로 포함하고 있지 않다. 그래서 =--with-tree-sitter= 옵션을 추가하여 소스 빌드해야 이 기능을 사용할 수 있다(더불어 소스 빌드 하는 김에 실행 속도를 높이기 위해 =--with-native-compilation== 옵션도 추가하는 것이 좋음).

특정 언어 코딩 시에 tree-sitter를 이용하기 위해서는 해당 언어의 lagnuage grammer가 별도로 필요한데, =M-x treesit-install-language-grammer= 명령을 이용하여 필요한 패키지를 설치할 수 있다. 설치 위치는 =~/.emacs.d/tree-sitter= 를 이용한다. 이것들은 소스로 부터 동적 라이브러리로 빌드하여 사용하므로 시스템에 빌드툴이 미리 설치되어 있어야 한다.

  ;; Helper function to find jdtls installation
  (defun my/find-jdtls-installation ()
    "Find jdtls installation directory and return configuration list.
  Searches in common locations: ~/.vscode/extensions, ~/.local/share/nvim/mason/packages/jdtls"
    (let* ((vscode-ext-dir (expand-file-name "~/.vscode/extensions"))
           (mason-jdtls-dir (expand-file-name "~/.local/share/nvim/mason/packages/jdtls"))

** PDF Viewer

doc-view는 기본 해상도(100) 변경이 필요함. Org Agenda 등 다른 버퍼가 doc-view 창을 빼앗지 않도록 =display-buffer-alist= 로 보호한다.

#+begin_src elisp
  (defun my/buffer-has-doc-view-p (window)
    "Return non-nil if WINDOW displays a doc-view buffer."
    (with-current-buffer (window-buffer window)
      (derived-mode-p 'doc-view-mode)))

  (defun my/display-buffer-avoid-doc-view (buffer alist)
    "Display BUFFER in a window, avoiding doc-view windows.
This is a display-buffer action function."
    (let ((start-window (selected-window))
          (window nil))
      ;; Try to find a non-doc-view window
      (walk-windows
       (lambda (w)
         (unless (or window (my/buffer-has-doc-view-p w))
           (setq window w)))
       nil t)
      ;; If all windows are doc-view, split the frame
      (if window
          (window--display-buffer buffer window 'reuse alist)
        (let ((new-window (split-window-sensibly start-window)))
          (when new-window
            (window--display-buffer buffer new-window 'window alist))))))

  ;; Org Agenda가 doc-view 창을 사용하지 않도록 설정
  (add-to-list 'display-buffer-alist
               '("\\*Org Agenda\\*"
                 (my/display-buffer-avoid-doc-view)))

  (use-package doc-view
    :custom
    (doc-view-resolution 200))
#+end_src

** Claude CLI Integration

Claude CLI를 Emacs와 통합하여 사용할 수 있는 기능들을 제공한다.
우선, Cluade CLI가 우선 [[https://docs.anthropic.com/en/docs/claude-code/setup][설치]]되어 있어야 한다. Linux/macOS에서는 =libvterm= 이 시스템에 설치되어 있어야 한다. Ubuntu에서는 =sudo apt install libvterm-dev= 명령으로 설치할 수 있다. Windows에서는 vterm이 지원되지 않으므로 shell을 사용한다.

#+begin_src emacs-lisp
  ;; vterm 패키지 설정 (Claude CLI 대화형 세션용) - Windows 제외
  (unless (eq system-type 'windows-nt)
    (use-package vterm
      :bind ("C-c t" . vterm)
      :config
      (setq vterm-max-scrollback 10000)
      (setq vterm-buffer-name-string "vterm %s")))

  ;; Claude CLI 대화형 세션 시작
  (defun claude-shell ()
    "Start Claude CLI in terminal"
    (interactive)
    (let ((claude-path (claude-executable-find)))
      (if (eq system-type 'windows-nt)
          ;; Windows: use Windows Terminal with MSYS2 bash
          ;; Set HOME explicitly to Windows user profile for consistent claude config
          (let* ((bash "C:\\msys64\\usr\\bin\\bash.exe")
                 (home (getenv "USERPROFILE"))
                 (cmd (format "HOME='%s' %s" home claude-path)))
            (call-process "cmd" nil nil nil "/c" "start" "wt" "-w" "0" "nt"
                          bash "-l" "-c" cmd))
        ;; Linux/macOS: use vterm
        (if (fboundp 'vterm)
            (progn
              (vterm)
              (vterm-send-string claude-path)
              (vterm-send-return))
          (user-error "vterm not available")))))

  ;; Claude CLI 실행 가능 여부 확인
  (defun claude-executable-find ()
    "Check if Claude CLI is available and return its path."
    (let ((claude-path (or (executable-find "claude")
                           ;; Check common installation paths
                           (seq-find #'file-executable-p
                                     '("/opt/node22/bin/claude"
                                       "~/.claude/local/claude"
                                       "~/.local/bin/claude"
                                       "~/bin/claude"
                                       "/usr/local/bin/claude")))))
      (unless claude-path
        (user-error "Claude CLI not found. Please install it from https://docs.anthropic.com/en/docs/claude-code/setup"))
      (expand-file-name claude-path)))

  ;; 선택된 영역을 Claude에 보내기
  (defun claude-send-region (start end)
    "Send selected region to Claude CLI"
    (interactive "r")
    (let ((claude-path (claude-executable-find))
          (text (buffer-substring-no-properties start end)))
      (with-temp-buffer
        (insert text)
        (let ((exit-code (call-process-region
                          (point-min) (point-max)
                          claude-path
                          t "*Claude Output*" nil
                          "--print")))
          (unless (zerop exit-code)
            (error "Claude CLI failed with exit code %d" exit-code))))
      (display-buffer "*Claude Output*")))

  ;; 전체 버퍼를 Claude에 보내기
  (defun claude-send-buffer ()
    "Send entire buffer to Claude CLI"
    (interactive)
    (claude-send-region (point-min) (point-max)))

  ;; Claude에게 질문하기
  (defun claude-ask (prompt)
    "Ask Claude a question and display result"
    (interactive "sAsk Claude: ")
    (let ((claude-path (claude-executable-find)))
      (with-current-buffer (get-buffer-create "*Claude Response*")
        (erase-buffer)
        (let ((exit-code (call-process claude-path nil t nil "--print" prompt)))
          (unless (zerop exit-code)
            (error "Claude CLI failed with exit code %d" exit-code)))
        (when (fboundp 'markdown-mode)
          (markdown-mode))
        (goto-char (point-min))
        (display-buffer (current-buffer)))))

  (defun my/shell-quote (str)
    "Quote STR for safe use in POSIX shell single-quoted strings.
Handles embedded single quotes by replacing ' with '\"'\"'."
    (concat "'" (replace-regexp-in-string "'" "'\"'\"'" str) "'"))

  ;; Start Claude session in project root
  (defun claude-project ()
    "Start Claude CLI in project root"
    (interactive)
    (let* ((claude-path (claude-executable-find))
           (project-dir (if (fboundp 'projectile-project-root)
                            (projectile-project-root)
                          default-directory)))
      (if (eq system-type 'windows-nt)
          ;; Windows: use Windows Terminal with project directory
          ;; Set HOME explicitly to Windows user profile for consistent claude config
          (let* ((bash "C:\\msys64\\usr\\bin\\bash.exe")
                 (home (getenv "USERPROFILE"))
                 (cmd (format "HOME=%s cd %s && %s"
                              (my/shell-quote home)
                              (my/shell-quote project-dir)
                              claude-path)))
            (call-process "cmd" nil nil nil "/c" "start" "wt" "-w" "0" "nt"
                          bash "-l" "-c" cmd))
        ;; Linux/macOS: use vterm
        (let ((default-directory project-dir))
          (if (fboundp 'vterm-other-window)
              (progn
                (vterm-other-window)
                (vterm-send-string claude-path)
                (vterm-send-return))
            (user-error "vterm not available"))))))

  ;; 키바인딩 설정
  ;; C-c A prefix for Claude/AI commands (A = AI/Assistant)
  (global-set-key (kbd "C-c A c") 'claude-shell)      ; C-c A c: start Claude shell
  (global-set-key (kbd "C-c A r") 'claude-send-region) ; C-c A r: send region
  (global-set-key (kbd "C-c A b") 'claude-send-buffer) ; C-c A b: send buffer
  (global-set-key (kbd "C-c A a") 'claude-ask)         ; C-c A a: ask question
  (global-set-key (kbd "C-c A p") 'claude-project)     ; C-c A p: project session
#+end_src
